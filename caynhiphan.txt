#include <bits/stdc++.h>
using namespace std;
void ptop(vector<int>& pre, int start, int en) {
    if (start > en) return;
    int root = pre[start];
    int idx = start + 1;

    while (idx <= en && pre[idx] < root) idx++;

    ptop(pre, start + 1, idx - 1);

    ptop(pre, idx, en);

    cout << root << " ";
}

int main() {
    int T;
    cin >> T;
    while (T--) {
        int N;
        cin >> N;
        vector<int> pre(N);
        for (int i = 0; i < N; i++) cin >> pre[i];

        ptop(pre, 0, N - 1);
        cout << "\n";
    }
}
