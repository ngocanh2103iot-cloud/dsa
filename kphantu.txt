#include <bits/stdc++.h>

using namespace std;
int n,a[1005],ok=1,k;
void ktao(){
    for(int i=1;i<=k;i++)
        a[i]=i;
}
void sinh(){
    int i=k;
    while(i>0 && a[i]==n-k+i) {
            i--;
    }
    if(i==0) ok=0;
    else{
        a[i]++;
        for(int j=i+1;j<=k;j++) a[j]=a[j-1]+1;
    }
}
int main()
{
    int t;cin>>t;
    while(t--){
    ok=1;
    cin>>n>>k;
    int b[n+1];
    for(int i=1;i<=n;i++) cin>>b[i];
    sort(b+1,b+n+1);
    ktao();
    while(ok){
        for(int i=1;i<=k;i++) cout<<b[a[i]]<<" ";
        cout<<endl;
        sinh();
    }
    cout<<endl;
}
        return 0;
}
